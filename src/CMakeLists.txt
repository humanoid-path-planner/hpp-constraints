  differentiable-function.cc
  generic-transformation.cc
  relative-com.cc
  com-between-feet.cc
  distance-between-bodies.cc
  distance-between-points-in-bodies.cc
  configuration-constraint.cc
  convex-shape-contact.cc
  static-stability.cc
  explicit-solver.cc
  hybrid-solver.cc
  iterative-solver.cc
)

  ADD_SUBDIRECTORY(_constraints)
